22  /***
23   * This class holds configuration details for Jawr in a given ServletContext. 
24   * 
25   * @author Jordi Hernández Sellés

28  public class JawrConfig {
29  	/***
30  	 * Initialize configuration using params contained in the initialization properties file. 
31  	 * @param configId
32  	 * @param props
33  	 */
34  	public JawrConfig(Properties props) 
35  	{
36  		
37  		if(null != props.getProperty("jawr.debug.on"))
38  		{			
39  			setDebugModeOn(Boolean.valueOf(props.getProperty("jawr.debug.on")).booleanValue());
40  		}	
41  		if(null != props.getProperty("jawr.gzip.on"))
42  		{			
43  			setGzipResourcesModeOn(Boolean.valueOf(props.getProperty("jawr.gzip.on")).booleanValue());
44  		}	
45  		if(null != props.getProperty("jawr.charset.name"))
46  		{			
47  			setCharsetName(props.getProperty("jawr.charset.name"));
48  		}	
49  		if(null != props.getProperty("jawr.gzip.ie6.on"))
50  		{			
51  			setGzipResourcesForIESixOn(Boolean.valueOf(props.getProperty("jawr.gzip.ie6.on")).booleanValue());
52  		}	
53  	}
54

58  	private String charsetName = "UTF-8";

107 	/***
108 	 * Get the charset to interpret and generate resource. 
109 	 * @return Charset
110 	 */
111 	public Charset getResourceCharset() {
112 		if(null == resourceCharset)
113 		{
114 			resourceCharset = Charset.forName(charsetName);
115 		}
116 		return resourceCharset;
117 	}
118 	
119 	/***
120 	 * Set the charsetname to be used to interpret and generate resource. 
121 	 * @param charsetName
122 	 */
123 	public void setCharsetName(String charsetName) {
124 		if(!Charset.isSupported(charsetName))
125 			throw new IllegalArgumentException("The specified charset [" + charsetName + "] is not supported by the jvm.");
126 		this.charsetName = charsetName;
127 	}

144 	}
