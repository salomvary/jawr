28  /***
29   * Registry for resource generators, which create scripts or CSS data dynamically, 
30   * as opposed to the usual behavior of reading a resource from the war file. 
31   * It provides methods to determine if a path mapping should be handled by a generator, 
32   * and to actually render the resource using the appropiate generator. 
33   * Path mappings which require generation will use a prefix (preferrably one which ends with 
34   * a colon, such as 'messages:'). 
35   * Generators provided with Jawr will be automatically mapped. 
36   * 
37   * @author Jordi Hernández Sellés
38   *
39   */
40  public class GeneratorRegistry {
41  	
42  	public static final String MESSAGE_BUNDLE_PREFIX = "messages:";
43  	public static final String CLASSPATH_BUNDLE_PREFIX = "jar:";
44  	public static final String DWR_BUNDLE_PREFIX = "dwr:";
45  	private static final Map registry = new HashMap();
46  	
47  	
48  	static
49  	{
50  		registry.put(MESSAGE_BUNDLE_PREFIX, new ResourceBundleMessagesGenerator());
51  		registry.put(CLASSPATH_BUNDLE_PREFIX, new ClasspathResourceGenerator());
52  	}
53  	
54  
55  	public GeneratorRegistry(ServletContext servletContext) {
56  		super();
57  		registry.put(DWR_BUNDLE_PREFIX, new DWRResourceGeneratorWrapper(servletContext));
58  	}
59  	
60  	/***
61  	 * Use only for testing purposes.
62  	 */
63  	public GeneratorRegistry(){
64  		super();
65  	}
66  	
67  	/***
68  	 * Register a generator mapping it to the specified prefix. 
69  	 * 
70  	 * @param prefix
71  	 * @param clazz
72  	
73  	public void registerGenerator(String prefix, Class clazz){
74  		
75  	} */
76  	
77  	/***
78  	 * Determines wether a path is to be handled by a generator. 
79  	 * @param path
80  	 * @return
81  	 */
82  	public boolean isPathGenerated(String path) {
83  		return null != matchPath(path);
84  	}
85  	
86  	/***
87  	 * Creates the contents corresponding to a path, by using the appropiate generator. 
88  	 * @param path
89  	 * @param charset
90  	 * @return
91  	 */
92  	public Reader createResource(String path,Charset charset) {
93  		String key = matchPath(path);
94  		return ((ResourceGenerator)registry.get(key)).createResource(path.substring(key.length()),charset);
95  	}
96  	
97  	/***
98  	 * Get the key from the mappings that corresponds to the specified path. 
99  	 * @param path
100 	 * @return
101 	 */
102 	private String matchPath(String path) {
103 		String rets = null;
104 		for(Iterator it = registry.keySet().iterator();it.hasNext() && rets == null;) {
105 			String prefix = (String) it.next();
106 			if(path.startsWith(prefix))
107 				rets = prefix;			
108 		}	
109 		return rets;
110 	}
