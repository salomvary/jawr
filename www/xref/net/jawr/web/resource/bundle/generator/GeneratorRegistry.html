32  /***
33   * Registry for resource generators, which create scripts or CSS data dynamically, 
34   * as opposed to the usual behavior of reading a resource from the war file. 
35   * It provides methods to determine if a path mapping should be handled by a generator, 
36   * and to actually render the resource using the appropiate generator. 
37   * Path mappings which require generation will use a prefix (preferrably one which ends with 
38   * a colon, such as 'messages:'). 
39   * Generators provided with Jawr will be automatically mapped. 
40   * 
41   * @author Jordi Hernández Sellés
42   *
43   */
44  public class GeneratorRegistry {
45  	
46  	public static final String MESSAGE_BUNDLE_PREFIX = "messages:";
47  	public static final String CLASSPATH_BUNDLE_PREFIX = "jar:";
48  	public static final String DWR_BUNDLE_PREFIX = "dwr:";
49  	public static final String COMMONS_VALIDATOR_PREFIX = "acv:";
50  	
51  	private static final Map registry = ConcurrentCollectionsFactory.buildConcurrentHashMap();
52  	private static final List prefixRegistry = ConcurrentCollectionsFactory.buildCopyOnWriteArrayList();
53  	private ServletContext servletContext;
54  	private JawrConfig config;
55  	
56  	static
57  	{
58  		prefixRegistry.add(MESSAGE_BUNDLE_PREFIX);
59  		prefixRegistry.add(CLASSPATH_BUNDLE_PREFIX);
60  		prefixRegistry.add(DWR_BUNDLE_PREFIX);
61  		prefixRegistry.add(COMMONS_VALIDATOR_PREFIX);
62  	}
63

65  	public GeneratorRegistry(ServletContext servletContext) {
66  		super();
67  		this.servletContext = servletContext;
68  	}

71  	 * Use only for testing purposes.
72  	 */
73  	public GeneratorRegistry(){

77  	/***
78  	 * Lazy loads generators, to avoid the need for undesired dependencies. 
79  	 * 
80  	 * @param generatorKey
81  	 * @return
82  	 */
83  	private void loadGenerator(String generatorKey) {
84  		if(MESSAGE_BUNDLE_PREFIX.equals(generatorKey)){
85  			registry.put(generatorKey, new ResourceBundleMessagesGenerator());
86  		}
87  		else if(CLASSPATH_BUNDLE_PREFIX.equals(generatorKey)){
88  			registry.put(generatorKey, new ClasspathResourceGenerator());
89  		}
90  		else if(DWR_BUNDLE_PREFIX.equals(generatorKey)){
91  			registry.put(generatorKey, DWRGeneratorFactory.createDWRGenerator());
92  		}
93  		else if(COMMONS_VALIDATOR_PREFIX.equals(generatorKey)){
94  			registry.put(generatorKey, new CommonsValidatorGenerator());
95  		}
96  	}

108 	/***
109 	 * Determines wether a path is to be handled by a generator. 
110 	 * @param path
111 	 * @return
112 	 */
113 	public boolean isPathGenerated(String path) {
114 		return null != matchPath(path);
115 	}

117 	/***
118 	 * Creates the contents corresponding to a path, by using the appropiate generator. 
119 	 * @param path
120 	 * @param charset
121 	 * @return
122 	 */
123 	public Reader createResource(String path, Charset charset) {
124 		String key = matchPath(path);
125 		Locale locale = null;
126 		if(path.indexOf('@') != -1){
127 			String localeKey = path.substring(path.indexOf('@')+1);
128 			path = path.substring(0,path.indexOf('@'));
129 			
130 			// Resourcebundle should be doing this for me...
131 			String[] params = localeKey.split("_");			
132 			switch(params.length) {
133 				case 3:
134 					locale = new Locale(params[0],params[1],params[2]);
135 					break;
136 				case 2: 
137 					locale = new Locale(params[0],params[1]);
138 					break;
139 				default:
140 					locale = new Locale(localeKey);
141 			}
142 		}
143 		return ((ResourceGenerator)registry.get(key)).createResource(path.substring(key.length()),config,servletContext,locale,charset);
144 	}
145 	
146 	/***
147 	 * Get the key from the mappings that corresponds to the specified path. 
148 	 * @param path
149 	 * @return
150 	 */
151 	private String matchPath(String path) {
152 		String rets = null;
153 		for(Iterator it = prefixRegistry.iterator();it.hasNext() && rets == null;) {
154 			String prefix = (String) it.next();
155 			if(path.startsWith(prefix))
156 				rets = prefix;			
157 		}	
158 		// Lazy load generator
159 		if(null != rets && !registry.containsKey(rets))
160 			loadGenerator(rets);
161 		
162 		return rets;
163 	}
