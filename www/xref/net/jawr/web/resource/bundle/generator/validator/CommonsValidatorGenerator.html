72  		locale = null == locale ? Locale.getDefault() : locale;
73  		String[] params = GeneratorParamUtils.getParenthesesParam(path, null);
74  		String validators = params[0];
75  		String validatorParams = params[1];
76  		
77  		String messagesNS = MessageBundleScriptCreator.DEFAULT_NAMESPACE;
78  		boolean stopOnErrors = true;
79  		
80  		StringTokenizer valparams = new StringTokenizer(validatorParams,"|");
81  		String validatorMappingKey = valparams.nextToken();
82  		if(valparams.hasMoreTokens())
83  			messagesNS = valparams.nextToken();
84  		if(valparams.hasMoreTokens())
85  			stopOnErrors = Boolean.valueOf(valparams.nextToken()).booleanValue();
86  		
87  		
88  		// Build the ValidatorResources if it doesn't exist yet
89  		if(!validatorResourcesMap.containsKey(validatorMappingKey)){			
90  			createValidatorResources(validatorMappingKey, config, servletContext);
91  		}
92  		
93  		ValidatorResources validatorResources = (ValidatorResources) validatorResourcesMap.get(validatorMappingKey);
94  		StringBuffer sb = new StringBuffer();
95  		StringTokenizer tk = new StringTokenizer(validators,"|");
96  		while(tk.hasMoreTokens()){
97  			String validator = tk.nextToken();
98  			if(STATIC_JAVASCRIPT_KEY.equals(validator)){
99  				sb.append(buildStaticJavascript(validatorResources));
100 			}
101 			else sb.append(buildFormJavascript(validatorResources, validator, locale,messagesNS,stopOnErrors));
102 			
103 		}
104 
105         return new StringReader(sb.toString() );
106 	}

120 	 private StringBuffer createDynamicJavascript(ValidatorResources resources, Form form, String messageNS, boolean stopOnErrors) {
121         StringBuffer results = new StringBuffer();
122 
123         List actions = this.createActionList(resources, form);
124 
125         final String methods =
126             this.createMethods(actions,stopOnErrors);
127 
128         String jsFormName = form.getName();
129 
130 
131         results.append(this.getJavascriptBegin(jsFormName,methods));
132 
133         for (Iterator i = actions.iterator(); i.hasNext();) {
134             ValidatorAction va = (ValidatorAction) i.next();
135             int jscriptVar = 0;
136             String functionName = null;
137 
138             if ((va.getJsFunctionName() != null)
139                 && (va.getJsFunctionName().length() > 0)) {
140                 functionName = va.getJsFunctionName();
141             } else {
142                 functionName = va.getName();
143             }
144 
145             results.append("    function " + jsFormName + "_" + functionName
146                 + " () { \n");
147 
148             for (Iterator x = form.getFields().iterator(); x.hasNext();) {
149                 Field field = (Field) x.next();
150 
151                 // Skip indexed fields for now until there is a good way to
152                 // handle error messages (and the length of the list (could
153                 // retrieve from scope?))
154                 if (field.isIndexed() || !field.isDependency(va.getName())) {
155                     continue;
156                 }
157 
158                 String message = null;
159                 Msg msg = field.getMessage(va.getName());
160                 if ((msg != null) && !msg.isResource()) {
161                 	message = JavascriptStringUtil.quote(msg.toString());
162                 }
163                 else{
164                 	if (msg == null) {
165                 		message = va.getMsg();
166                     } else {
167                     	message = msg.getKey();
168                     }
169                 	Arg[] args = field.getArgs(va.getName());
170                 	
171                 	message = messageNS + "." + message + "(";
172                 	for (int a = 0; a < args.length; a++) {
173                 		if (args[a] != null) {
174                 			if (args[a].isResource()) {
175                 				message += messageNS + "." + args[a].getKey() + "()";
176                             } else {
177                             	message += "\"" + args[a].getKey() + "\"";
178                             }
179 	                		message += ",";
180                 		}
181                 	}
182                 	message+= "null)";
183                 }
184 
185                 message = (message != null) ? message : "";
186                 
187                 // prefix variable with 'a' to make it a legal identifier
188                 results.append("     this.a" + jscriptVar++ + " = [\""
189                     + field.getKey() + "\", " + (message)
190                     + ", ");
191 
192                 results.append("function(varName){");
193 
194                 Map vars = field.getVars();
195 
196                 // Loop through the field's variables.
197                 Iterator varsIterator = vars.keySet().iterator();
198 
199                 while (varsIterator.hasNext()) {
200                     String varName = (String) varsIterator.next();
201                     Var var = (Var) vars.get(varName);
202                     String varValue = var.getValue();
203 
204                     // Non-resource variable
205                     if (var.isResource()) {
206                     	varValue = messageNS + "." + varValue + "()";
207                     }
208                     else  varValue ="'"+ varValue + "'";
209                     String jsType = var.getJsType();
210 
211                     // skip requiredif variables field, fieldIndexed, fieldTest,
212                     // fieldValue
213                     if (varName.startsWith("field")) {
214                         continue;
215                     }
216 
217                     String varValueEscaped = JavascriptStringUtil.escape(varValue);
218 
219                     if (Var.JSTYPE_INT.equalsIgnoreCase(jsType)) {
220                         results.append("this." + varName + "=+"
221                             + varValueEscaped + "; ");
222                     } else if (Var.JSTYPE_REGEXP.equalsIgnoreCase(jsType)) {
223                         results.append("this." + varName + "=eval('/'+"
224                             + varValueEscaped + "+'/'); ");
225                     } else if (Var.JSTYPE_STRING.equalsIgnoreCase(jsType)) {
226                         results.append("this." + varName + "="
227                             + varValueEscaped + "; ");
228 
229                         // So everyone using the latest format doesn't need to
230                         // change their xml files immediately.
231                     } else if ("mask".equalsIgnoreCase(varName)) {
232                         results.append("this." + varName + "=eval('/'+"
233                             + varValueEscaped + "+'/'); ");
234                     } else {
235                         results.append("this." + varName + "="
236                             + varValueEscaped + "; ");
237                     }
238                 }
239 
240                 results.append(" return this[varName];}];\n");
241             }
242 
243             results.append("    } \n\n");
244         }
245         return results;
246     }

248  	/***
249      * Returns the opening script element and some initial javascript.
250      */
251     protected String getJavascriptBegin(String jsFormName,String methods) {
252         StringBuffer sb = new StringBuffer();
253         String name = jsFormName.replace('/', '_'); // remove any '/' characters
254 
255         name =
256             jsFormName.substring(0, 1).toUpperCase()
257             + jsFormName.substring(1, jsFormName.length());
258 
259         sb.append("\n    var bCancel = false; \n\n");
260         sb.append("    function validate" + name + "(form) { \n");
261         sb.append("        if (bCancel) { \n");
262         sb.append("            return true; \n");
263         sb.append("        } else { \n");
264 
265         // Always return true if there aren't any Javascript validation methods
266         if ((methods == null) || (methods.length() == 0)) {
267             sb.append("            return true; \n");
268         } else {
269             sb.append("            var formValidationResult; \n");
270             sb.append("            formValidationResult = " + methods + "; \n");
271             if (methods.indexOf("&&") >= 0) {
272                 sb.append("            return (formValidationResult); \n");
273             } else {
274                 //Making Sure that Bitwise operator works:
275                 sb.append("            return (formValidationResult == 1); \n");
276             }
277         }
278         sb.append("        } \n");
279         sb.append("    } \n\n");
280 
281         return sb.toString();
282     }

284 	private List createActionList(ValidatorResources resources, Form form) {
285         List actionMethods = new ArrayList();
286 
287         Iterator iterator = form.getFields().iterator();
288 
289         while (iterator.hasNext()) {
290             Field field = (Field) iterator.next();
291 
292             for (Iterator x = field.getDependencyList().iterator();
293                 x.hasNext();) {
294                 Object o = x.next();
295 
296                 if ((o != null) && !actionMethods.contains(o)) {
297                     actionMethods.add(o);
298                 }
299             }
300         }
301 
302         List actions = new ArrayList();
303 
304         // Create list of ValidatorActions based on actionMethods
305         iterator = actionMethods.iterator();
306 
307         while (iterator.hasNext()) {
308             String depends = (String) iterator.next();
309             ValidatorAction va = resources.getValidatorAction(depends);
310 
311             // throw nicer NPE for easier debugging
312             if (va == null) {
313                 throw new NullPointerException("Depends string \"" + depends
314                     + "\" was not found in validator-rules.xml.");
315             }
316 
317             if ((va.getJavascript() != null)
318                 && (va.getJavascript().length() > 0)) {
319                 actions.add(va);
320             } else {
321                 iterator.remove();
322             }
323         }
324 
325         Collections.sort(actions, actionComparator);
326 
327         return actions;
328     }

330 	private String createMethods(List actions, boolean stopOnErrors) {
331         StringBuffer methods = new StringBuffer();
332         final String methodOperator = stopOnErrors ? " && " : " & ";
333 
334         Iterator iter = actions.iterator();
335 
336         while (iter.hasNext()) {
337             ValidatorAction va = (ValidatorAction) iter.next();
338 
339             if (methods.length() > 0) {
340                 methods.append(methodOperator);
341             }
342 
343             methods.append(va.getMethod()).append("(form)");
344         }
345 
346         return methods.toString();
347     }

349 	private StringBuffer buildStaticJavascript(ValidatorResources validatorResources) {
350 		StringBuffer sb = new StringBuffer();
351 
352         Iterator actions = validatorResources.getValidatorActions().values().iterator();
353 
354         while (actions.hasNext()) {
355             ValidatorAction va = (ValidatorAction) actions.next();
356 
357             if (va != null) {
358                 String javascript = va.getJavascript();
359 
360                 if ((javascript != null) && (javascript.length() > 0)) {
361                     sb.append(javascript).append("\n");
362                 }
363             }
364         }
365 
366         return sb;
367 	}

369 	private void createValidatorResources(String path, JawrConfig config, ServletContext servletContext) {
370 		
371 		ValidatorResources validatorResources=null;
372 		String configPaths = config.getConfigProperties().getProperty(path);
373 		
374 		StringTokenizer st = new StringTokenizer(configPaths, ",");
375         InputStream[] inputStreams = new InputStream[st.countTokens()];
376         int pos = 0;
377         
378         try {
379             while (st.hasMoreTokens()) {
380                 String validatorRules = st.nextToken().trim();
381 
382                 if (log.isDebugEnabled()) {
383                     log.debug("Validation rules file from '"
384                         + validatorRules + "'");
385                 }
386                 InputStream is = null;
387                 try {
388                 	is = ClassLoaderResourceUtils.getResourceAsStream(validatorRules, this);
389                 }
390                 catch(FileNotFoundException fos) {
391                 	is = servletContext.getResourceAsStream(validatorRules);
392                 }
393                 inputStreams[pos] = is;
394                 pos++;
395             }
396             validatorResources = new ValidatorResources(inputStreams);            
397         } catch (SAXException ex) {
398             throw new RuntimeException(ex);
399         } catch (IOException e) {
400             throw new RuntimeException(e);
401 		} 
402 		validatorResourcesMap.put(path, validatorResources);
403 	}

405 	private static final Comparator actionComparator =
406         new Comparator() {
407             public int compare(Object o1, Object o2) {
408                 ValidatorAction va1 = (ValidatorAction) o1;
409                 ValidatorAction va2 = (ValidatorAction) o2;
410 
411                 if (((va1.getDepends() == null)
412                     || (va1.getDepends().length() == 0))
413                     && ((va2.getDepends() == null)
414                     || (va2.getDepends().length() == 0))) {
415                     return 0;
416                 } else if (((va1.getDepends() != null)
417                     && (va1.getDepends().length() > 0))
418                     && ((va2.getDepends() == null)
419                     || (va2.getDepends().length() == 0))) {
420                     return 1;
421                 } else if (((va1.getDepends() == null)
422                     || (va1.getDepends().length() == 0))
423                     && ((va2.getDepends() != null)
424                     && (va2.getDepends().length() > 0))) {
425                     return -1;
426                 } else {
427                     return va1.getDependencyList().size()
428                     - va2.getDependencyList().size();
429                 }
430             }
431         };
