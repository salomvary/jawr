28  /***
29   * Single file postprocessor used to rewrite CSS URLs according to the new relative locations of the references when
30   * added to a bundle. Since the path changes, the URLs must be rewritten accordingly.  
31   * URLs in css files are expected to be according to the css spec (see http://www.w3.org/TR/REC-CSS2/syndata.html#value-def-uri). 
32   * Thus, single double, or no quotes enclosing the url are allowed (and remain as they are after rewriting). Escaped parens and quotes 
33   * are allowed within the url.   
34   *  
35   * @author Jordi Hernández Sellés
36   *
37   */
38  public class CSSURLPathRewriterPostProcessor extends
39  		AbstractChainedResourceBundlePostProcessor {
40  	
41  	private static final Pattern urlPattern = Pattern.compile(	"url//(//s*" // 'url(' and any number of whitespaces 
42  																+ "((//////))|[^)])*" // any sequence of characters, except an unescaped ')'
43  																+ "//s*//)",  // Any number of whitespaces, then ')'
44  																Pattern.CASE_INSENSITIVE); // works with 'URL('
45  	private static final String backRefRegex = "(//.//./)";
46  	private static final Pattern backRefRegexPattern = Pattern.compile(backRefRegex);
47  	private static final Pattern URLSeparatorPattern = Pattern.compile("/");
48  	
49  	/* (non-Javadoc)
50  	 * @see net.jawr.web.resource.bundle.postprocess.impl.AbstractChainedResourceBundlePostProcessor#doPostProcessBundle(net.jawr.web.resource.bundle.postprocess.BundleProcessingStatus, java.lang.StringBuffer)
51  	 */
52  	protected StringBuffer doPostProcessBundle(BundleProcessingStatus status,
53  			StringBuffer bundleData) throws IOException {
54  		
55  		String data = bundleData.toString();
56  		
57  		// Initial backrefs (number of backtracking paths needed, i.e. '../').
58  		int initialBackRefs = 1;
59  		
60  		if(! "".equals(status.getJawrConfig().getServletMapping())){
61  			StringTokenizer tk = new StringTokenizer(status.getJawrConfig().getServletMapping(),"/");
62  			initialBackRefs += tk.countTokens();
63  		}
64  		
65  		String bundleName = status.getCurrentBundle().getName();
66  		initialBackRefs += numberOfForwardReferences(bundleName);
67  		List resourceBackRefs = getPathNamesInResource(status.getLastPathAdded());
68  		Matcher matcher = urlPattern.matcher(data);
69  		StringBuffer sb = new StringBuffer();
70  		while(matcher.find()) {
71  			List backRefs = new ArrayList();
72  			backRefs.addAll(resourceBackRefs);
73  			String url = getUrlPath(matcher.group(), initialBackRefs, backRefs, status.getJawrConfig().getCssImagePathOverride());
74  			matcher.appendReplacement(sb, RegexUtil.adaptReplacementToMatcher(url));
75  		}
76  		matcher.appendTail(sb);
77  		return sb;
78  	}

81  	/***
82  	 * Transform a matched url so it points to the proper relative path with respect to the given path.  
83  	 * @param matcher
84  	 * @param initialBackRefs
85  	 * @return
86  	 */
87  	private String getUrlPath(String match, int initialBackRefs, List resourceBackRefs, String imagePathOverride) {
88  
89  		String url = match.substring(match.indexOf('(')+1,match.lastIndexOf(')'))
90  					.trim();
91  
92  		// To keep quotes as they are, first they are checked and removed. 
93  		String quoteStr = "";
94  		if( url.startsWith("'") || url.startsWith("\"")) {
95  			quoteStr = url.charAt(0)+"";
96  			url = url.substring(1,url.length()-1);
97  		}
98  		
99  		// Check if the URL is absolute, if it is return it as is. 
100 		int firstSlash = url.indexOf('/');
101 		if(0 == firstSlash || (firstSlash != -1 && url.charAt(++firstSlash) == '/')){
102 			StringBuffer sb = new StringBuffer("url(");
103 			sb.append(quoteStr).append(url).append(quoteStr).append(")");
104 			return sb.toString();
105 		}
106 		
107 		int backRefsInURL = 0;
108 		// Remove leading slash
109 		if(url.startsWith("../")) {
110 			Matcher backUrls = backRefRegexPattern.matcher(url);
111 			while(backUrls.find())
112 				backRefsInURL++;
113 			url = url.replaceAll(backRefRegex, "");
114 		}
115 		else if(url.startsWith("/"))
116 			url = url.substring(1,url.length());
117 		else if(url.startsWith("./"))
118 			url = url.substring(2,url.length());
119 		
120 		// append the override url here if need be
121 		if(imagePathOverride != null){
122 			StringBuffer sb = new StringBuffer("url(");
123 			sb.append(quoteStr).append(imagePathOverride).append(url).append(quoteStr).append(")");
124 			return sb.toString();
125 		}
126 		
127 		// Adjust the forward pathnames according to the backrefs in the url
128 		if(backRefsInURL > 0 && resourceBackRefs.size() > 0) {
129 			int size = resourceBackRefs.size();
130 			int counter = 0;
131 			while(size > 0) {
132 				counter++;
133 				if(counter >= 50)
134 					throw new RuntimeException("Infinite loop on url: " + url + " :size:" + size);
135 				size--;
136 				backRefsInURL--;
137 				resourceBackRefs.remove(size);
138 				if(backRefsInURL == 0)
139 					break;
140 			}
141 		}		
142 		
143 		// Start rendering the result, starting by the initial quote, if any. 
144 		StringBuffer urlPrefix = new StringBuffer("url(").append(quoteStr);
145 		
146 		// Add the back references as needed
147 		for (int i = 0; i < initialBackRefs; i++) {
148 			urlPrefix.append("../");
149 		}		
150 		for(Iterator it = resourceBackRefs.iterator();it.hasNext(); ) {
151 			urlPrefix.append(it.next()).append("/");			
152 		}
153 		return urlPrefix.append(url).append(quoteStr).append(")").toString();
154 	}

157 	/***
158 	 * Gets the path names within a reource URL, exluding the resource file name. 
159 	 * @param resourceURL
160 	 * @return
161 	 */
162 	private List getPathNamesInResource(String resourceURL) {
163 		List names = new ArrayList();
164 		String[] namesArray = resourceURL.split("/");
165 		
166 		// Add all but the last pathname
167 		for (int i = 0; i < namesArray.length -1 ; i++) {
168 			if(!"".equals(namesArray[i]))
169 				names.add(namesArray[i]);
170 		}
171 		
172 		return names;
173 	}
174 	
175 	/***
176 	 * Find the number of occurrences of / within a url. 
177 	 * @param url
178 	 * @return
179 	 */
180 	private int numberOfForwardReferences(String url) {
181 		Matcher backUrls = URLSeparatorPattern.matcher(url);
182 		int numMatches = 0;
183 		while(backUrls.find())
184 			numMatches++;
185 		
186 		if(url.startsWith("/"))
187 			numMatches--;
188 		
189 		return numMatches;
190 	}
